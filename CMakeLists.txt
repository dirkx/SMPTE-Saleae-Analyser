cmake_minimum_required (VERSION 3.13)

project(SMPTE-Saleae-Analyzer)

add_definitions( -DLOGIC2 )

# enable generation of compile_commands.json, helpful for IDEs to locate include files.
set(CMAKE_EXPORT_COMPILE_COMMANDS ON)

# custom CMake Modules are located in the cmake directory.
set(CMAKE_MODULE_PATH ${PROJECT_SOURCE_DIR}/cmake)

include(ExternalAnalyzerSDK)

set(SOURCES
	src/SMPTEAnalyzer.cpp
	src/SMPTEAnalyzerSettings.cpp
	src/SMPTEAnalyzer.h
	src/SMPTEAnalyzerSettings.h
	src/SMPTEAnalyzerResults.cpp
	src/SMPTESimulationDataGenerator.cpp
	src/SMPTEAnalyzerResults.h
	src/SMPTESimulationDataGenerator.h
)

add_analyzer_plugin(${PROJECT_NAME} SOURCES ${SOURCES})
